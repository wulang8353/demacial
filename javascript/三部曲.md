## 新手成长

- 理解DOM

```
<label>请输入北京今天空气质量：<input id="aqi-input" type="text"></label>
<button id="button">确认填写</button>
<div>您输入的值是：<span id="aqi-display">尚无录入</span></div>

<script type="text/javascript">
  (function() {
    /*    
    在注释下方写下代码
    给按钮button绑定一个点击事件
    在事件处理函数中
    获取aqi-input输入的值，并显示在aqi-display中
    */
    var btn = document.getElementById('button'); 取到按钮的引用
    var word = document.getElementById('aqi-input'); 取到输入框的引用
    var display = document.getElementById('aqi-display');  

    btn.onclick = function() {  使用DOM0级事件
        display.innerHTML = word.value;
    }
    
    或者DOM2级事件：
    btn.addEventListener("click",function() {
       display.innerHTML = word.value; 
    })
    
    改进：
    var num = parseInt(word.value.trim())
    if((!NaN(num) && (num>=0) && (num<=500)){
        display.innerHTML = word.value;
    } else if (input.value === '') {
        display.innerHTML = '空气污染指数不能为空';
      }
      else {
        display.innerHTML = '请输入一个0~1000的数字';
      }    
})();
</script>
```
